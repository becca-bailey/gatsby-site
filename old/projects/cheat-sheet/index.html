Object Constructor Methods


  function Rectangle(height, width) {
    this.height = height;
    this.width = width;
    this.calcArea = function() {
      return this.height * this.width
    }
    this.calcPerimeter = function() {
      return this.height * 2 + this.width * 2;
    }
  }

  var rect = new Rectangle(7, 3);
  var area = rect.calcArea();
  var perimeter = rect.calcPerimeter();
  console.log("Area = " + area);
  console.log("Perimeter = " + perimeter);

  => Area = 21
  => Perimeter = 20

Prototypes and Inheritance


  function Animal(name, numLegs) {
    this.name = name;
    this.numLegs = numLegs;
  }

  function Penguin(name) {
    this.name = name;
    this.numLegs = 2;
  }

  Penguin.prototype = new Animal();
  // Penguin is new instance of Animal
  var spot = new Penguin("Spot");
  console.log(spot.numLegs);

  => 2

Arrays

Create an Array


  cities = ["Chicago", "New York", "San Francisco"]


  cities = Array.new
  cities[0] = "Chicago"
  cities[1] = "New York",
  cities[2] = "San Francisco"

  numbers = %w[one two three]

Nested Arrays


  complement = [["red", "green"]["orange", "blue"]["purple", "yellow"]]


  complement.each do |sub_array|
    sub_array.each do |color|
      puts color
    end
  end

  => red
  => green
  => orange
  => blue
  => purple
  => yellow

Hashes


  # Literal Notation

  my_hash = { name: "Becca", 
              age: "27",
              location: "Chicago"
            }


  # Hash.new method

  my_hash = Hash.new
  my_hash["Becca"] = :name
  puts my_hash["Becca"]

  => :name

Accessing Elements


  puts my_hash[:name]

  => Becca

Iterating


  my_hash.each { |key, value| puts "#{key}: #{value}" }

  => name: Becca
  => age: 27
  => location: Chicago

Flow Control


  x = 10; y = 15
  x > y ? greatest = x : greatest = y
  puts "The greater number is #{greatest}."

  => "The greater number is 15."

Enumerables


  .all? # checks if condition is true for all
  %w[ant bear cat].all? { |word| word.length >= 3 }

  => true

  .any? # checks if condition is true for any
  .chunk # groups like items together
  .count # returns number of true instances
  ary = [1, 2, 3, 4]
  ary.count { |x| x % 2 === 0 }

  => 2

  .drop # drops items from an array
  ary = [1, 2, 3, 4, 5, 6]
  ary.drop(3)

  => [4, 5, 6]

  .reduce # performs an operation on each item
  (5..10).reduce(:+)

  => 45

  .inject # performs an operation on each item
  (5..10).inject { |sum, n| sum + n }

  => 45
